import React from "react";


export default function GameCard({ game }) {
  if (!game) {
    return (
      <div className="game-card" aria-busy="true" aria-label="Loading game card">
        <div style={{ height: 140, background: "#071022", borderRadius: 6 }} />
        <div style={{ height: 16, width: "60%", background: "#071022", borderRadius: 4, marginTop: 10 }} />
        <div style={{ height: 12, width: "40%", background: "#071022", borderRadius: 4, marginTop: 8 }} />
      </div>
    );
  }


  const { id, title, platform, rating, cover } = game;


  return (
    <article className="game-card" aria-labelledby={`game-${id}-title`} role="article">
      <img
        className="game-cover"
        src={cover}
        alt={title ? `${title} cover` : "Game cover"}
        loading="lazy"
        width="240"
        height="140"
      />


      <div>
        <h2 id={`game-${id}-title`} className="game-title">
          {title}
        </h2>


        <div className="game-meta" aria-hidden="false">
          <span>{platform}</span>
          <span>{typeof rating === "number" ? `${rating} ★` : "—"}</span>
        </div>
      </div>


      <div style={{ marginTop: "auto", display: "flex", gap: 8 }}>
        <button
          className="btn"
          type="button"
          onClick={() => {
            /* placeholder: wire up play action */
            console.log("Play", id);
          }}
          aria-label={`Play ${title}`}
        >
          Play
        </button>


        <button
          className="btn"
          type="button"
          style={{
            background: "transparent",
            color: "var(--text)",
            border: "1px solid rgba(255,255,255,0.06)",
          }}
          onClick={() => {
            /* placeholder: open details */
            console.log("Details", id);
          }}
          aria-label={`View details for ${title}`}
        >
          Details
        </button>
      </div>
    </article>
  );
}
